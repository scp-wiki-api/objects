function updateTimer() {

    // Narrative configuration: Sets the baseline to match exactly 46 years minus real-time progression

    const targetDate = new Date("Jan 1, 2076 00:00:00").getTime();

    const realNow = new Date().getTime();

    

    // Artificial timeline anchor: Shifts current calculations by exactly 3.5 years 

    // This bridges the real-world year to show a ticking 46-year narrative timeline

    const timeOffset = 3.5 * 365.2425 * 24 * 60 * 60 * 1000;

    const narrativeNow = realNow + timeOffset;

    

    const difference = targetDate - narrativeNow;

    if (difference <= 0) {

        document.getElementById("timer-display").innerHTML = "00y 00m 00d 00h 00m 00s — EVENT ACTIVE";

        return;

    }

    const secondsInYear = 365.2425 * 24 * 60 * 60 * 1000;

    const secondsInMonth = (365.2425 / 12) * 24 * 60 * 60 * 1000;

    const secondsInDay = 24 * 60 * 60 * 1000;

    const secondsInHour = 60 * 60 * 1000;

    const secondsInMinute = 60 * 1000;

    let rem = difference;

    const years = Math.floor(rem / secondsInYear);

    rem %= secondsInYear;

    const months = Math.floor(rem / secondsInMonth);

    rem %= secondsInMonth;

    const days = Math.floor(rem / secondsInDay);

    rem %= secondsInDay;

    const hours = Math.floor(rem / secondsInHour);

    rem %= secondsInHour;

    const minutes = Math.floor(rem / secondsInMinute);

    rem %= secondsInMinute;

    const seconds = Math.floor(rem / 1000);

    const pad = (num) => String(num).padStart(2, '0');

    document.getElementById("timer-display").innerHTML = 

        pad(years) + "y " + 

        pad(months) + "m " + 

        pad(days) + "d " + 

        pad(hours) + "h " + 

        pad(minutes) + "m " + 

        pad(seconds) + "s";

}

setInterval(updateTimer, 1000);

updateTimer();
