function initNightSky() {

  const canvas = document.getElementById('sky-canvas');

  if (!canvas) return;

  const deskBg = document.getElementById('desk-bg');

  const W = deskBg ? deskBg.offsetWidth  : 900;

  const H = deskBg ? deskBg.offsetHeight : 450;

  canvas.width  = W;

  canvas.height = H;

  const ctx = canvas.getContext('2d');

  const STAR_COUNT = 320;

  const stars = Array.from({length: STAR_COUNT}, () => {

    const big = Math.random() < 0.12;  // 12% are large prominent stars

    return {

      x:     Math.random() * W,

      y:     Math.random() * H,

      r:     big ? 1.2 + Math.random() * 1.6 : 0.2 + Math.random() * 0.7,

      base:  big ? 0.55 + Math.random() * 0.45 : 0.1 + Math.random() * 0.4,

      phase: Math.random() * Math.PI * 2,

      speed: big ? 0.6 + Math.random() * 1.2 : 2 + Math.random() * 5,

      amp:   big ? 0.5 : 0.25,  // twinkle amplitude

      big,

      glitch: false,

      glitchTimer: 0,

    };

  });

  const persistTears = [];

  let tearCooldown = 30;

  let burstActive  = false;   // multi-tear burst mode

  let burstCount   = 0;

  let noiseTimer   = 8;

  let flashTimer   = 60;

  function draw(ts) {

    ctx.clearRect(0, 0, W, H);

    const grad = ctx.createLinearGradient(0, 0, 0, H);

    grad.addColorStop(0,   '#000000');

    grad.addColorStop(0.5, '#010101');

    grad.addColorStop(1,   '#000000');

    ctx.fillStyle = grad;

    ctx.fillRect(0, 0, W, H);

    const vig = ctx.createRadialGradient(W/2, H*0.45, H*0.05, W/2, H*0.45, H*0.9);

    vig.addColorStop(0, 'rgba(0,0,0,0)');

    vig.addColorStop(1, 'rgba(0,0,0,0.65)');

    ctx.fillStyle = vig;

    ctx.fillRect(0, 0, W, H);

    const t = ts * 0.001;

    stars.forEach(s => {

      const glitchChance = s.big ? 0.0006 : 0.0002;

      if (!s.glitch && Math.random() < glitchChance) {

        s.glitch = true;

        s.glitchTimer = s.big ? 12 + Math.floor(Math.random()*25) : 5 + Math.floor(Math.random()*12);

      }

      let alpha, drawX = s.x, drawY = s.y, drawR = s.r;

      if (s.glitch) {

        s.glitchTimer--;

        if (s.glitchTimer <= 0) { s.glitch = false; }

        const gPct = 1 - s.glitchTimer / 25;

        drawX += (Math.random() - 0.5) * (s.big ? 14 : 6);

        drawY += (Math.random() - 0.5) * (s.big ? 4 : 2);

        alpha  = Math.random() < 0.45 ? 0 : 0.7 + Math.random() * 0.3;

        drawR  = s.r * (0.3 + Math.random() * 3.5);

        if (Math.random() < 0.55) {

          const smearW = (s.big ? 8 : 3) + Math.random() * (s.big ? 50 : 20);

          const smearX = drawX + (Math.random() - 0.5) * 30;

          ctx.fillStyle = `rgba(255,255,255,${0.06 + Math.random() * (s.big ? 0.2 : 0.08)})`;

          ctx.fillRect(smearX, drawY, smearW, s.big ? 2 : 1);

        }

      } else {

        alpha = s.base * (1 - s.amp + s.amp * Math.sin(t * s.speed + s.phase));

      }

      if (s.big && !s.glitch && alpha > 0.3) {

        const gr = drawR * 5;

        const glow = ctx.createRadialGradient(drawX, drawY, 0, drawX, drawY, gr);

        glow.addColorStop(0,   `rgba(255,255,255,${alpha * 0.22})`);

        glow.addColorStop(0.4, `rgba(255,255,255,${alpha * 0.08})`);

        glow.addColorStop(1,   'rgba(255,255,255,0)');

        ctx.fillStyle = glow;

        ctx.fillRect(drawX - gr, drawY - gr, gr*2, gr*2);

function handleL3Choice(entry) {

  if (l3Typing) return;

  document.querySelectorAll('.l3-choice').forEach(b => b.classList.add('used'));

  l3Visited.add(entry.id);

  l3Exchanges++;

  const question = document.getElementById('l3-question');

  if (question) {

    question.textContent = '» ' + entry.label;

    question.classList.remove('visible');

    setTimeout(() => question.classList.add('visible'), 30);

  }

  if (entry.unlocks && entry.unlocks > l3Stage) {

    l3Stage = entry.unlocks;

  }

  if (l3Exchanges >= 5 && l3Stage < 4) l3Stage = 4;

  setTimeout(() => {

    if (entry.lines.some(l => l.text === '**END**')) {

      showL3Lines(entry.lines.filter(l => l.text !== '**END**'), endL3);

      return;

    }

    showL3Lines(entry.lines, () => buildL3Choices());

  }, 450);

}

function eraseL3Line(el, callback) {

  if (!el || !el.parentNode) { if (callback) callback(); return; }

  const full = el.textContent;

  const isSymbol = el.classList.contains('symbol');

  if (isSymbol) {

    el.classList.add('fading');

    setTimeout(() => { el.remove(); if (callback) callback(); }, 380);

    return;

  }

  let len = full.length;

  function tick() {

    if (len <= 0) {

      el.classList.add('fading');

      setTimeout(() => { el.remove(); if (callback) callback(); }, 200);

      return;

    }

    len -= 1 + (Math.random() < 0.2 ? 1 : 0); // occasionally erase 2 at once

    el.textContent = full.slice(0, Math.max(0, len));

    setTimeout(tick, 22 + (Math.random() < 0.1 ? 60 : 0));

  }

  tick();

}

function typeL3Line(text, type, callback) {

  const output = document.getElementById('l3-output');

  const el = document.createElement('div');

  el.className = 'l3-line' + (type ? ' ' + type : '');

  output.appendChild(el);

  if (type === 'symbol') {

    el.textContent = text;

    setTimeout(() => el.classList.add('visible'), 40);

    if (callback) setTimeout(() => callback(el), 200);

    return;

  }

  el.classList.add('visible');

  let ci = 0;

  const baseSpeed = (type === 'entity') ? 52 : 36;

  function tick() {

    if (ci >= text.length) {

      if (callback) callback(el);

      return;

    }

    if (Math.random() < 0.025 && ci > 4 && type === 'entity') {

      const g = '▓░▒';

      el.textContent = text.slice(0, ci) + g[Math.floor(Math.random() * g.length)];

      setTimeout(() => { el.textContent = text.slice(0, ci); setTimeout(tick, baseSpeed); }, 80);

      return;

    }

    el.textContent = text.slice(0, ++ci);

    const ch = text[ci - 1];

    const pause = (ch === ',' || ch === ';') ? 180 : (ch === '.' || ch === '?') ? 280 : 0;

    setTimeout(tick, baseSpeed + pause + (Math.random() < 0.06 ? 140 : 0));

  }

  tick();

}
