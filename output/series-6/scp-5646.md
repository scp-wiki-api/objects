var START_INTERVAL    = 22000; 

var FIRST_OFFSET_MIN  = 15000; 

var FIRST_OFFSET_MAX  = 27000;

var FAST_TAU_DOWN     = 6000;

var THRESH_START      = 0.90;

var THRESH_FLOOR      = 0.06;

var THRESH_CEIL       = 0.90;

var THRESH_DROP       = 0.05;

var THRESH_UNIT       = 0.05;

var THRESH_WEIGHT_BIAS= 0.09;

var globalSlips = 0;

function clamp(v, lo, hi){ return v < lo ? lo : (v > hi ? hi : v); }

function Field(label, slipFn, weight, opts){

  this.label     = label;

  this.slip      = slipFn;

  this.weight    = weight || 1;

  this.fast      = !!(opts && opts.fast);

  this.bias      = (this.weight - 1) * THRESH_WEIGHT_BIAS;

  this.floor     = this.fast ? FAST_FLOOR

                             : Math.min(MAX_INTERVAL, MIN_INTERVAL * this.weight * this.weight);

  this.base      = START_INTERVAL + (this.weight - 1) * WEIGHT_BASE_STEP;

  this.threshold = THRESH_START;

  this.interval  = this.base;

  this.lag       = 0;  

  this.acc       = 0; 

  this.lastBeat  = 0; 

}

Field.prototype.beat = function(){

  var rippled = globalSlips - this.seen;

  this.seen = globalSlips;

  try {

    if (Math.random() > this.threshold + this.bias) {

      this.slip();

      globalSlips++;

      rippled++;

    }

  } catch (e) { }

  if (this.phase === 'decohere') {

    this.threshold -= THRESH_DROP;

    this.interval  -= rippled * INTERVAL_STEP / this.weight;

    this.interval   = clamp(this.interval, this.floor, MAX_INTERVAL);

    if (this.threshold <= THRESH_FLOOR) {

      this.threshold = THRESH_FLOOR;

      this.phase = 'cohere';

      this.fibA = 1; this.fibB = 1;

    }

  } else {

    this.threshold += (0.5 * rippled * THRESH_UNIT) + BASE_CREEP;

    var fibNext = this.fibA + this.fibB;

    this.fibA = this.fibB; this.fibB = fibNext;

    this.interval += fibNext * INTERVAL_STEP;

    this.interval  = clamp(this.interval, this.floor, MAX_INTERVAL);

    if (this.threshold >= THRESH_CEIL) {

      this.threshold = THRESH_START;

      this.interval  = this.base;

      this.phase = 'decohere';

    }

  }

  var jitter = 0.85 + Math.random() * 0.30;

  var next = Math.max(this.floor, this.interval * jitter);

  if (!this.fast) {

    var span = this.base - this.floor;

    var pace = span > 0 ? clamp(1 - (this.interval - this.floor) / span, 0, 1) : 1;

    next += Math.random() * STAGGER_MAX * pace;

  }

  var self = this;

  setTimeout(function(){ self.beat(); }, next);

};

Field.prototype.fastBeat = function(){

  var now = (Date.now ? Date.now() : new Date().getTime());

  var dt = this.lastBeat ? (now - this.lastBeat) : FAST_TICK;

  this.lastBeat = now;

  var swing = THRESH_START - THRESH_FLOOR;

  if (this.phase === 'decohere') {

    this.threshold -= swing * dt / FAST_INHALE_MS;

    if (this.threshold <= THRESH_FLOOR) { this.threshold = THRESH_FLOOR; this.phase = 'cohere'; }

  } else {

    this.threshold += swing * dt / FAST_EXHALE_MS;

    if (this.threshold >= THRESH_CEIL) { this.threshold = THRESH_START; this.phase = 'decohere'; }

  }

  var b = (THRESH_START - this.threshold) / swing;      

  var tau = (b > this.lag) ? FAST_TAU_UP : FAST_TAU_DOWN; 

  this.lag += (b - this.lag) * (1 - Math.exp(-dt / tau));  

  this.lag = clamp(this.lag, 0, 1);

 

  var changeInterval = FAST_FLOOR * Math.pow(MAX_INTERVAL / FAST_FLOOR, 1 - this.lag);

  this.acc += dt;

  if (this.acc >= changeInterval) {

    this.acc = 0;

    try { this.slip(); } catch (e) { }

  }

  var self = this;

  setTimeout(function(){ self.fastBeat(); }, FAST_TICK);

};

<p><strong>ITEM #:</strong> SCP-<span class="numba1">5</span><span class="numba2">6</span><span class="numba3">4</span><span class="numba4">6</span></p>

<p><strong>SPECIAL CONTAINMENT PROCEDURES:</strong> SCP-<span class="numba1">5</span><span class="numba2">6</span><span class="numba3">4</span><span class="numba4">6</span><span ID="contain"> is not to remain open for passage, nor should it be closed or obstructed in a way that would prevent passage into the anomaly.</span></p>

<p><strong>DESCRIPTION:</strong> SCP-<span class="numba1">5</span><span class="numba2">6</span><span class="numba3">4</span><span class="numba4">6</span> is a <span class="descstruct">hallway</span> in Site-<span ID="site2">64</span>, sublevel <span ID="level">27</span>. SCP-<span class="numba1">5</span><span class="numba2">6</span><span class="numba3">4</span><span class="numba4">6</span>'s destination is unknown, but as of the events of Incident <span class="numba1">5</span><span class="numba2">6</span><span class="numba3">4</span><span class="numba4">6</span>-A, documentation regarding the anomaly is subject to extreme narrativistic decoherence, suggesting <a href="https://scp-wiki.wikidot.com/sandrewswann-s-proposal">[DATA EXPUNGED]</a>.</p>

<p><strong><span class="sublastname">Halliburton</span>:</strong> I feel fine—just don't see what all the fuss is about. You didn't need to bring out the whole cavalcade just for me.</p>

<p><strong><span class="sublastname">Halliburton</span>:</strong> Is this because of the sign? Look, I didn't realize it was a big issue, the phrasing is kind of confusing. I thought it led to the commissary—at least, it is the right direction. I would know; I've had to carry a compass just to get around here without being late to everything.</p>

<p><strong><span class="seclastname">Dougherty</span>:</strong> [pause] <span class="subfirstname">Gene</span>, are you aware of the events of the last few weeks? Specifically, what happened when you were on PTO?</p>

<p><strong><span class="sublastname">Halliburton</span>:</strong> Wha—of course I am. Is this a comment about my mental acuity? I'd remind you that jokes aren't jokes if they hurt feelings and we are in a harassment-free workplace. [pause] It was <span class="descrelation">my daughter's birthday</span>, <span class="secfirstname">Hane</span>, I wouldnt forget that. <span class="descouting">We went to the beach.</span> I took the next few days off and went to Vegas with the guys. Lost a couple hundred at the casino since I left my wallet next to the vending machines. Most expensive bag of <span class="descsnack">Combos</span> I've ever had.</p>

<p><strong><span class="seclastname">Dougherty</span>:</strong> And what about <span class="descstate">Arizona</span>? You must remember that.</p>

<p><strong><span class="sublastname">Halliburton</span>:</strong> Yeah? What about it? It was a nice way to round off the trip. Weather was lovely. Got to see the <span class="descvortex">vortex</span>, which was almost certainly anomalous, by the way. We should probably look into that.</p>

<p><strong><span class="seclastname">Dougherty</span>:</strong> I guess so. Well, while you were gone, Logistics updated the keycard readers and issued new codes. The old ones stop working throughout the day, so I'll need to update yours immediately if you don't want to get flagged every time you do so much as wash your hands.</p>

<p><strong><span class="sublastname">Halliburton</span>:</strong> Ah, I guess that explains things.</p>

<p><em>Some rustling is heard.</em></p>

<p><strong><span class="sublastname">Halliburton</span>:</strong> Here you go.</p>

<p><strong><span class="seclastname">Dougherty</span>:</strong> Perfect, thank you. Just be a sec.</p>

<p><em>A few seconds of silence, followed by the sound of sliding plastic, ending on a series of out-of-key beeps.</em></p>

<p><strong><span class="seclastname">Dougherty</span>:</strong> Hmmm.</p>

<p><em>Another plastic noise, followed by more flat tones.</em></p>

<p><strong><span class="sublastname">Halliburton</span>:</strong> Is there a problem?</p>

<p><strong><span class="seclastname">Dougherty</span>:</strong> It's probably this shoddy clearance handheld, they have spotty connections down here, I apologize. I'll get it done in a snap, hold tight.</p>

<p><em>Scraping and creaking as a chair is pushed back. <span class="seclastname">Dougherty</span> is heard getting up and leaving the room.</em></p>

<p><strong><span class="sublastname">Halliburton</span>:</strong> [sighs]</p>

<p><em>Roughly two minutes of silence pass.</em></p>

<p><em>The door is heard opening, followed by a several pairs of footsteps.</em></p>

<p><strong>Security Personnel:</strong> Alright, let's go! Up! Now!</p>

<p><strong><span class="sublastname">Halliburton</span>:</strong> What—?</p>

<p><em>Sounds of commotion, struggling.</em></p>

<p><strong><span class="sublastname">Halliburton</span>:</strong> No, look at my badge, I—I'm a level <span ID="clearancespell">three</span>—[grunt]—fuck—LEVEL <span ID="clearancespellcap">THREE</span>! <span class="deptshort">R&D</span>!—[groan]—</p>

<p><strong>Security Personnel:</strong> UP! When I say <em>let's go</em>, you <em>go</em>!</p>

<p><strong><span class="sublastname">Halliburton</span>:</strong>—But—agh—I-I didn't do anything! [yelps]—I <em>didn't</em>—<em>do anything</em>!</p>

<p><em>Thirty seconds of struggle, accompanied by a dragging sound.</em></p>

<p><strong><span class="sublastname">Halliburton</span>:</strong> H-HELP! Somebody help!—Please!—What are you going to do to me?—[sobs]—Let me go! Please, I want to see my family! <span class="lovedone">M-MAGGIE</span>!—[sobs]</p>

<p><em>Microphone records similar for the next few minutes. The commotion steadily grows quiet, and someone still in the room shuts off tape shortly thereafter.</em></p>
